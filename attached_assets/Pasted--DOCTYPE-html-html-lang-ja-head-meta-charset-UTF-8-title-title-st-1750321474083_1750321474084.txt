    <script>
        function bookmarkAnswer() {
            const answer = document.querySelector('.model-answer').innerText;
            let bookmarks = JSON.parse(localStorage.getItem('bookmarks')) || [];
            if (!bookmarks.includes(answer)) {
                bookmarks.push(answer);
                localStorage.setItem('bookmarks', JSON.stringify(bookmarks));
                alert('ブックマークに追加しました！');
            } else {
                alert('すでにブックマークされています。');
            }
        }
    </script>
